        ddCommon
        devdriver
)

    inc/ddEventParserApi.h
    inc/ddEventParser.h
    src/ddEventParser.cpp

    src/eventParser.h
    src/eventParser.cpp

    src/legacy/legacyEventParser.h
    src/legacy/legacyEventParser.cpp
)
